83 @_catch_exception 

84 def latest_message( 

85 self, 

86 sender: str, 

87 date_range: int = -1, 

88 message_parts: str = TEXT_MESSSAGE, 

89 ) -> tuple[str, _AnyResponseData]: 

90 """Convenience method for retrieving the latest message from a 

91 specified sender. 

92 

93 Use the largest UID to get the most recent message. Since the search 

94 key `ON` command cannot guarantee a result, this method uses 

95 `SENTSINCE` and the days prior today —represented by negative 

96 integers. 

97 

98 Args: 

99 sender (str): String contained in the envelope structure's FROM 

100 field. 

101 date_range (int, optional): Time frame or days in which a message 

102 is expected to be present. Defaults to -1. 

103 message_parts (str, optional): Message data item names. Defaults 

104 to TEXT_MESSSAGE (RFC822/BODY[]). 

105 

106 Raises: 

107 Exception: Raised when date_range is greater than -1. 

108 Exception: No message was found from specified sender. 

109 

110 Returns: 

111 tuple[str, _AnyResponseData]: IMAP response type and the message 

112 data. 

113 """ 

114 if date_range > -1: 

115 raise Exception( 

116 f"date_range can't be greater than -1: \ 

117 {date_range}" 

118 ) 

119 

120 latest_uid = 0 

121 sentsince = date_format(date_travel(date_range)) 

122 src, dat = self._decode_search_res( 

123 self.imap4.search(None, f'(FROM "{sender}")')[1], 

124 self.imap4.search(None, f"(SENTSINCE {sentsince})")[1], 

125 ) 

126 

127 for uid in dat[0].split(" "): 

128 _uid = int(uid) 

129 if uid in src[0].split(" "): 

130 # Since UIDs are incremented, its length is relative to 

131 # the message's freshness; the larger the UID, the more 

132 # recent the message. 

133 if _uid > latest_uid: 

134 latest_uid = _uid 

135 

136 if latest_uid > 0: 

137 return self.imap4.fetch(str(latest_uid), message_parts) 

138 

139 raise Exception(f"Message not found: {latest_uid}")

179 Args:

194 if port == 993:
